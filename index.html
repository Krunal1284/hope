      startFlowers();

    },800);

  },500);

}


/* screen navigation */

function goToScreen(n){

  document.querySelectorAll('.screen').forEach(s=>s.classList.remove('active'));

  if(n<=2){

    const sc = document.getElementById('screen'+n);

    if(sc) sc.classList.add('active');

    window.scrollTo({top:0,behavior:'smooth'});

  } else {

    showEnding();

  }

}


/* side flowers */

const flLayer=document.getElementById('flower-layer');

function startFlowers(){

  const fl=['🌸','🌷','🌺'];

  setInterval(()=>{

    const p=document.createElement('span');

    const left=Math.random()>0.5;

    const x=left?Math.random()*10:90+Math.random()*9;

    const dur=5+Math.random()*5;

    p.style.cssText=`position:fixed;left:${x}%;top:-20px;font-size:${11+Math.random()*10}px;pointer-events:none;animation:sidefall ${dur}s linear forwards;opacity:0;`;

    p.textContent=fl[Math.floor(Math.random()*fl.length)];

    flLayer.appendChild(p);

    setTimeout(()=>p.remove(),(dur*1000)+300);

  },600);

}


/* ending */

function showEnding(){

  document.querySelectorAll('.screen').forEach(s=>s.classList.remove('active'));

  document.getElementById('scene-end').classList.add('show');

  window.scrollTo({top:0,behavior:'smooth'});

  const rain=document.getElementById('end-rain');

  const items=['🥹','🌸','😭','🌷','💌','🌺','🌹','😭'];

  for(let i=0;i<55;i++){

    setTimeout(()=>{

      const e=document.createElement('span');

      const dur=4+Math.random()*5;

      e.className='ri';

      e.style.cssText=`left:${Math.random()*100}%;font-size:${16+Math.random()*28}px;animation-duration:${dur}s;animation-delay:${Math.random()*3}s;`;

      e.textContent=items[Math.floor(Math.random()*items.length)];

      rain.appendChild(e);

    },i*55);

  }

}
